536 inline byte PS2X::getCorrectedAxis(byte const joystickAxis) {
537  if (!axisCentered(joystickAxis)) {
538  byte coord = Analog(joystickAxis);
539  if (m_LXtolerance.lower != 0 || m_LYtolerance.lower != 0 || m_RXtolerance.lower != 0 || m_RYtolerance.lower != 0) {
540  switch (joystickAxis) {
541  case PSS_LX:
542  if (coord < m_LXtolerance.lower)
543  return map(coord, m_LXtolerance.lower, 0, 0, 255);
544  else
545  return map(coord, m_LXtolerance.upper, 255, 0, 255);
546  break;
547  case PSS_LY:
548  if (coord < m_LYtolerance.lower)
549  return map(coord, m_LYtolerance.lower, 0, 0, 255);
550  else
551  return map(coord, m_LYtolerance.upper, 255, 0, 255);
552  break;
553  case PSS_RX:
554  if (coord < m_RXtolerance.lower)
555  return map(coord, m_RXtolerance.lower, 0, 0, 255);
556  else
557  return map(coord, m_RXtolerance.upper, 255, 0, 255);
558  break;
559  case PSS_RY:
560  if (coord < m_RYtolerance.lower)
561  return map(coord, m_RYtolerance.lower, 0, 0, 255);
562  else
563  return map(coord, m_RYtolerance.upper, 255, 0, 255);
564  break;
565  }
566  }
567  else if (m_globalTolerance != 0) {
568  if (coord < 127)
569  return map(coord, 127 - m_globalTolerance, 0, 0, 255);
570  else
571  return map(coord, 127 + m_globalTolerance, 255, 0, 255);
572  }
573  else
574  return coord;
575  }
576 
577  return 0;
578 }

581 inline boolean PS2X::axisCentered(byte axis) {
582  if (m_LXtolerance.lower != 0 || m_LYtolerance.lower != 0 || m_RXtolerance.lower != 0 || m_RYtolerance.lower != 0) {
583  switch (axis) {
584  case PSS_LX:
585  if (Analog(PSS_LX) > m_LXtolerance.lower && Analog(PSS_LX) < m_LXtolerance.upper)
586  return true;
587  break;
588  case PSS_LY:
589  if (Analog(PSS_LY) > m_LYtolerance.lower && Analog(PSS_LY) < m_LYtolerance.upper)
590  return true;
591  break;
592  case PSS_RX:
593  if (Analog(PSS_RX) > m_RXtolerance.lower && Analog(PSS_RX) < m_RXtolerance.upper)
594  return true;
595  break;
596  case PSS_RY:
597  if (Analog(PSS_RY) > m_RYtolerance.lower && Analog(PSS_RY) < m_RYtolerance.upper)
598  return true;
599  break;
600  }
601  }
602  else if (m_globalTolerance != 0) {
603  return (Analog(axis) > 127 - m_globalTolerance && Analog(axis) < 127 + m_globalTolerance);
604  }
605  else
606  return (Analog(axis) == 127);
607  return false;
608 }
